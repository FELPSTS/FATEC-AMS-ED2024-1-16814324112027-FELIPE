Copy code
#include <stdio.h>

#define MAX 100

typedef struct {
    int queue[MAX];
    int front, rear;
} Queue;

void enqueue(Queue *q, int item) {
    if (q->rear == MAX - 1) {
        printf("Fila cheia\n");
        return;
    }
    q->queue[++q->rear] = item;
}

int dequeue(Queue *q) {
    if (q->front > q->rear) {
        printf("Fila vazia\n");
        return -1;
    }
    return q->queue[q->front++];
}

int main() {
    Queue q = {{}, -1, -1};
    
    enqueue(&q, 10);
    enqueue(&q, 20);
    enqueue(&q, 30);
    
    printf("%d\n", dequeue(&q)); // Saída: 10
    printf("%d\n", dequeue(&q)); // Saída: 20
    
    return 0;
}

int main() {
    int num = 10;

    printf("Conteúdo de num: %d\n", num);
    printf("Endereço de memória de num: %p\n", &num);
    printf("Conteúdo de ptr: %p\n", ptr);
    printf("Conteúdo apontado por ptr: %d\n", *ptr);

}

c
Copy code
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node *next;
} Node;

Node *createNode(int data) {
    Node *newNode = (Node *)malloc(sizeof(Node));
    newNode->data = data;
    newNode->next = NULL;
    return newNode;
}

int main() {
    Node *head = createNode(1);
    head->next = createNode(2);
    head->next->next = createNode(3);
    
    Node *current = head;
    while (current != NULL) {
        printf("%d ", current->data);
        current = current->next;
    }
    
    return 0;
}
